    Interface IStreamingManager

    IStreamingManager

    Description

    Defines the contract for the StreamingManager service. This service is responsible for creating, managing, and terminating data streams, as well as handling client subscriptions and data distribution.

    interface IStreamingManager {
        closeStream(streamId: string, reason?: string): Promise<void>;
        createStream(requestedStreamId?: string): Promise<string>;
        deregisterClient(streamId: string, clientId: string): Promise<void>;
        getActiveStreamIds(): Promise<string[]>;
        getClientCountForStream(streamId: string): Promise<number>;
        handleStreamError(
            streamId: string,
            error: Error,
            terminateStream?: boolean,
        ): Promise<void>;
        initialize(config: StreamingManagerConfig): Promise<void>;
        pushChunk(streamId: string, chunk: AgentOSResponse): Promise<void>;
        registerClient(streamId: string, client: IStreamClient): Promise<void>;
        shutdown(): Promise<void>;
    }
